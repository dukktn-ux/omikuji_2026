        function drawOmikuji() {
            const fortunes = ["大吉", "中吉", "小吉", "吉", "末吉", "凶"];
            // 乱数を使って結果を選択
            const index = Math.floor(Math.random() * fortunes.length);
            document.getElementById("result").innerText = fortunes[index];
        }
